50 def responders_to_json(responders: List, responder_key: str, one_is_dict: bool = False) \ 

51 -> Dict[str, Union[List, Dict]]: # type: ignore 

52 """ 

53 :param responders: the responders list which we get from demisto.args() 

54 :param responder_key: Some of the api calls need "responder" and others "responders" as a 

55 key in the responders jason 

56 :param one_is_dict: Some of the api calls need when there is one responder it as a dict 

57 and others as a list 

58 :return json_responders: reformatted respondres dict 

59 """ 

60 if not responders: 

61 return {} 

62 if len(responders) % 3 != 0: 

63 raise DemistoException("responders must be list of: responder_type, value_type, value") 

64 responders_triple = list(zip(responders[::3], responders[1::3], responders[2::3])) 

65 json_responders = {responder_key: []} # type: ignore 

66 for responder_type, value_type, value in responders_triple: 

67 if responder_type == "user" and value_type == "name": 

68 value_type = "username" 

69 json_responders[responder_key].append({value_type: value, "type": responder_type}) 

70 response = json_responders 

71 if len(responders_triple) == 1 and one_is_dict: 

72 response = {responder_key: json_responders[responder_key][0]} 

73 return response # type: ignore

235 @staticmethod 

236 def build_query(args: dict) -> str: 

237 query = "" 

238 if args.get("query", ""): 

239 query = args.get("query", "") 

240 if args.get("is_fetch_query", False) or not args.get("query", ""): 

241 status = args.get("status", ALL_TYPE) 

242 if status != ALL_TYPE: 

243 query += ' AND ' if query else '' 

244 query += f'status={status.lower()}' 

245 priority = argToList(args.get("priority", [ALL_TYPE])) 

246 if ALL_TYPE not in priority: 

247 query += ' AND ' if query else '' 

248 priority_parsed = ' OR '.join([p for p in priority]) 

249 query += f'priority: ({priority_parsed})' 

250 tags = argToList(args.get("tags", [])) 

251 if tags: 

252 query += ' AND ' if query else '' 

253 tag_parsed = ' OR '.join([t for t in tags]) 

254 query += f'tag: ({tag_parsed})' 

255 return query

322def run_polling_paging_command(args: dict, cmd: str, results_function: Callable, 

323 action_function: Optional[Callable] = None) -> CommandResults: 

324 

325 ScheduledCommand.raise_error_if_not_supported() 

326 

327 interval_in_secs = int(args.get('interval_in_seconds', 5)) 

328 result = args.get('result', []) 

329 limit = int(args.get('limit', 20)) 

330 

331 if "request_id" not in args and action_function: 

332 # starting new flow 

333 results = action_function(args) 

334 request_id = results.get("requestId") 

335 if not request_id: 

336 raise ConnectionError(f"Failed to send request - {results}") 

337 next_paging = results.get("paging", {}).get("next") 

338 result = result + results.get("data") 

339 if not next_paging or len(result) >= limit: 

340 # If not a paged request, simply return 

341 return CommandResults( 

342 outputs_prefix=args.get("output_prefix", "OpsGenie"), 

343 outputs=results.get("data"), 

344 readable_output=tableToMarkdown("OpsGenie", results.get('data'), 

345 headers=['id', 'createdAt', 'acknowledged', 'count', 'status', 'tags'], 

346 removeNull=True 

347 ), 

348 raw_response=results 

349 ) 

350 else: 

351 # If a paged request, return scheduled_command 

352 args['request_id'] = request_id 

353 args['result'] = result 

354 args['paging'] = next_paging 

355 polling_args = { 

356 'interval_in_seconds': interval_in_secs, 

357 'polling': True, 

358 **args 

359 } 

360 scheduled_command = ScheduledCommand( 

361 command=cmd, 

362 next_run_in_seconds=int(args.get('interval_in_seconds', 5)), 

363 args=polling_args, 

364 timeout_in_seconds=int(args.get('timeout_in_seconds', 60)), 

365 ) 

366 return CommandResults(scheduled_command=scheduled_command, 

367 readable_output=f"Waiting for request_id={request_id}", 

368 outputs_prefix=args.get("output_prefix", "OpsGenie"), 

369 outputs={"requestId": request_id}) 

370 

371 results = results_function(args) 

372 result = result + results.get("data") 

373 results['data'] = result 

374 next_paging = results.get("paging", {}).get("next") 

375 if not next_paging or len(result) >= limit: 

376 # If not a paged request, simply return 

377 return CommandResults( 

378 outputs_prefix=args.get("output_prefix", "OpsGenie"), 

379 outputs=results.get("data"), 

380 readable_output=tableToMarkdown("OpsGenie", results.get('data'), 

381 headers=['id', 'createdAt', 'acknowledged', 'count', 'status', 'tags'], 

382 removeNull=True 

383 ), 

384 raw_response=results 

385 ) 

386 

387 if len(result) < limit: 

388 # schedule next poll 

389 args['request_id'] = results.get('request_id') 

390 args['result'] = result 

391 args['paging'] = next_paging 

392 polling_args = { 

393 'interval_in_seconds': interval_in_secs, 

394 'polling': True, 

395 **args 

396 } 

397 scheduled_command = ScheduledCommand( 

398 command=cmd, 

399 next_run_in_seconds=int(args.get('interval_in_seconds', 5)), 

400 args=polling_args, 

401 timeout_in_seconds=int(args.get('timeout_in_seconds', 60)) 

402 ) 

403 # result with scheduled_command only - no update to the war room 

404 command_results = CommandResults(scheduled_command=scheduled_command, 

405 readable_output=f"Waiting for request_id={args.get('request_id')}", 

406 outputs_prefix=args.get("output_prefix", "OpsGenie"), 

407 outputs={"requestId": args.get('request_id')}) 

408 return command_results 

409 return CommandResults(outputs_prefix=args.get("output_prefix", "OpsGenie"), 

410 outputs=results.get("data"), 

411 readable_output=tableToMarkdown("OpsGenie", results.get('data'),

912def fetch_incidents_by_type(client: Client, 

913 params: Dict[str, Any], 

914 incident_fetching_func: Callable, 

915 now: datetime, 

916 last_run_dict: Optional[dict] = None): 

917 query = params.get('query') 

918 limit = int(params.get('max_fetch', 50)) 

919 fetch_time = params.get('first_fetch', '3 days').strip() 

920 

921 if not last_run_dict: 

922 new_last_run = _parse_fetch_time(fetch_time) 

923 last_run_dict = {'lastRun': new_last_run, 

924 'next_page': None} 

925 

926 if last_run_dict.get('next_page'): 

927 raw_response = client.get_paged({"paging": last_run_dict.get('next_page')}) 

928 else: 

929 timestamp_now = int(now.timestamp()) 

930 last_run = last_run_dict.get('lastRun') 

931 timestamp_last_run = int(dateparser.parse(last_run).timestamp()) 

932 time_query = f'createdAt>{timestamp_last_run} AND createdAt<={timestamp_now}' 

933 params['query'] = f'{query} AND {time_query}' if query else f'{time_query}' 

934 params['limit'] = limit 

935 params['is_fetch_query'] = True if params.get('query') else False 

936 raw_response = incident_fetching_func(params) 

937 last_run_dict['lastRun'] = now.strftime(DATE_FORMAT) 

938 

939 data = raw_response.get('data') 

940 incidents = [] 

941 if data: 

942 for event in data: 

943 incidents.append({ 

944 'name': event.get('message'), 

945 'occurred': event.get('createdAt'), 

946 'rawJSON': json.dumps(event) 

947 }) 

948 if last_run_dict.get('lastRun') < event.get('createdAt'): 

949 last_run_dict['lastRun'] = event.get('createdAt') 

950 

951 return incidents, raw_response.get("paging", {}).get("next"), last_run_dict.get('lastRun')

958def fetch_incidents_command(client: Client, 

959 params: Dict[str, Any], 

960 last_run: Optional[dict] = None) -> Tuple[List[Dict[str, Any]], Dict]: 

961 """Uses to fetch incidents into Demisto 

962 Documentation: https://github.com/demisto/content/tree/master/docs/fetching_incidents 

963 

964 Args: 

965 client: Client object with request 

966 last_run: Last fetch object occurs 

967 params: demisto params 

968 

969 Returns: 

970 incidents, new last_run 

971 """ 

972 demisto.debug(f"Got incidentType={params.get('event_types')}") 

973 event_type = params.get('event_types', [ALL_TYPE]) 

974 demisto.debug(f"Got event_type={event_type}") 

975 now = _get_utc_now() 

976 incidents = [] 

977 alerts = [] 

978 last_run_alerts = demisto.get(last_run, f"{ALERT_TYPE}.lastRun") 

979 next_page_alerts = demisto.get(last_run, f"{ALERT_TYPE}.next_page") 

980 last_run_incidents = demisto.get(last_run, f"{INCIDENT_TYPE}.lastRun") 

981 next_page_incidents = demisto.get(last_run, f"{INCIDENT_TYPE}.next_page") 

982 if ALERT_TYPE in event_type or ALL_TYPE in event_type: 

983 alerts, next_page_alerts, last_run_alerts = fetch_incidents_by_type(client, 

984 params, 

985 client.list_alerts, 

986 now, 

987 demisto.get(last_run, f"{ALERT_TYPE}")) 

988 if INCIDENT_TYPE in event_type or ALL_TYPE in event_type: 

989 incidents, next_page_incidents, last_run_incidents = fetch_incidents_by_type(client, 

990 params, 

991 client.list_incidents, 

992 now, 

993 demisto.get(last_run, f"{INCIDENT_TYPE}")) 

994 return incidents + alerts, {ALERT_TYPE: {'lastRun': last_run_alerts, 

995 'next_page': next_page_alerts}, 

996 INCIDENT_TYPE: {'lastRun': last_run_incidents, 

997 'next_page': next_page_incidents}
